        class NCR{
            static final long MOD = 1000000007;
            static final int MAX = 100001;
            static final long fact[] = new long[MAX];
            static final long invFact[] = new long[MAX];
            static{
                fact[0] = invFact[0] = 1;
                for(int i=1;i<MAX;i++){
                    fact[i] = (fact[i-1]*i)%MOD;
                }
                invFact[MAX - 1] = pow(fact[MAX - 1], MOD - 2);
                for(int i=MAX - 2;i>=1;i--){
                    invFact[i] = (invFact[i+1]*(i+1))%MOD;
                }
            }

            public static long pow(long a, long b){
                a %= MOD;
                long res = 1;
                while(b > 0){
                    if((b & 1) == 1) res = (res * a)%MOD;
                    a = (a * a)%MOD;
                    b >>= 1;
                }
                return res;
            }

            public long nCr(int n, int r){
                if(r < 0 || r > n) return 0;
                return (((fact[n] * invFact[r]) % MOD) * invFact[n - r]) % MOD;
            }
        }

        class MatrixExponentiation{
            long [][] multiply(long A[][], long B[][], int MOD){
                int n = A.length;
                long res[][] = new long[n][n];
                for(int i=0;i<n;i++){
                    for(int j=0;j<n;j++){
                        for(int k=0;k<n;k++){
                            res[i][j] = (res[i][j] + A[i][k]*B[k][j])%MOD;
                        }
                    }
                }
                return res;
            }

            long[][] matrixPower(long A[][], long power, int MOD){
                int n = A.length;
                long[][] res = new long[n][n];
                for(int i=0;i<n;i++) res[i][i] = 1;
                while(power > 0){
                    if((power & 1) == 1){
                        res = multiply(A, res, MOD);
                    }
                    A = multiply(A, A, MOD);
                    power >>= 1;
                }

                return res;
            }
        }
